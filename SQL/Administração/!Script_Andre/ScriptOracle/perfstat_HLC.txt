Cursor c_periodo_hn (pinicio varchar2,pfim varchar2) IS
  SELECT snap_id,snap_time
  FROM 
     stats$snapshot   
  WHERE 
     snap_time >= to_date(pinicio,'dd/mm/yyyy') and  snap_time < (to_date(pfim,'dd/mm/yyyy')+1/12)
ORDER BY snap_time;

Function Cal_Hit_Lib_Cache(fim_snap_id number, inicio_snap_id number) 
 Return number Is
   cursor LH (i_snap_id number) is
            select sum(pins), sum(pinhits)
              from stats$librarycache
             where snap_id         = i_snap_id;
--               and dbid            = p_dbid
--               and instance_number = p_instance_number;
   bpsum number;
   bhsum number;
   epsum number;
   ehsum number;
   vresult number;
   begin
      if not LH%ISOPEN then open LH (inicio_snap_id); end if;
      fetch LH into bpsum, bhsum;
      if LH%NOTFOUND then
        raise_application_error(-20100,'Missing start value for stats$librarycache');
      end if; close LH;
      if not LH%ISOPEN then open LH (fim_snap_id); end if;
      fetch LH into epsum, ehsum;
      if LH%NOTFOUND then
         raise_application_error(-20100,'Missing end value for stats$librarycache');
      end if; close LH;
      vresult := (ehsum - bhsum) / (epsum - bpsum);
      return (round(vresult,2));
end;
