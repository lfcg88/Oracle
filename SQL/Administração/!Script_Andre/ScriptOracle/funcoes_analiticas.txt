SELECT SQL_TEXT, PERC_DISK_READ * 100,PERC_BUFFER_GETS * 100
FROM ( SELECT SQL_TEXT,
        RATIO_TO_REPORT(DISK_READS) OVER ( ) PERC_DISK_READ,
        RATIO_TO_REPORT(BUFFER_GETS) OVER ( ) PERC_BUFFER_GETS  
        FROM PERFPRD1.STATS$SQL_SUMMARY
        WHERE SNAP_ID=6838
        ORDER BY 3 DESC )
