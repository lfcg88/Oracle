============
SAP_PROD_IG
============

CREATE ROLE RL_SAP_PROD_IG_S;

declare
i number;

begin
  for i in (select table_name from all_tables where OWNER = 'SAP_PROD_IG') loop
    execute immediate 'GRANT SELECT ON SAP_PROD_IG.'||I.TABLE_NAME||' TO RL_SAP_PROD_IG_S';
  end loop;
end;

declare
i number;

begin
  for i in (select VIEW_NAME from all_views where OWNER = 'SAP_PROD_IG') loop
    execute immediate 'GRANT SELECT ON SAP_PROD_IG.'||I.VIEW_NAME||' TO RL_SAP_PROD_IG_S';
  end loop;
end;

CREATE ROLE RL_SAP_PROD_IG_SU;

declare
i number;

begin
  for i in (select table_name from all_tables where OWNER = 'SAP_PROD_IG') loop
    execute immediate 'GRANT SELECT, UPDATE ON SAP_PROD_IG.'||I.TABLE_NAME||' TO RL_SAP_PROD_IG_SU';
  end loop;
end;

CREATE ROLE RL_SAP_PROD_IG_SIU;

declare
i number;

begin

    execute immediate 'GRANT SELECT, INSERT, UPDATE ON SAP_PROD_IG.'||I.TABLE_NAME||' TO RL_SAP_PROD_IG_SIU';
  end loop;
end;
