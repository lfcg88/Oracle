DROP USER SIGINPI CASCADE;

CREATE USER SIGINPI
  IDENTIFIED BY <password>
  DEFAULT TABLESPACE SIGINPI_TAB
  TEMPORARY TABLESPACE TEMP
  PROFILE DEFAULT
  ACCOUNT UNLOCK;
  -- 5 Roles for SIGINPI 
  GRANT CONNECT TO SIGINPI;
  GRANT RESOURCE TO SIGINPI;
  GRANT RL_CONSULTA_SIGINPI TO SIGINPI WITH ADMIN OPTION;
  GRANT RL_LINKDATA TO SIGINPI WITH ADMIN OPTION;
  GRANT RL_SIGINPI TO SIGINPI WITH ADMIN OPTION;
  ALTER USER SIGINPI DEFAULT ROLE ALL;
  -- 4 System Privileges for SIGINPI 
  GRANT CREATE ROLE TO SIGINPI;
  GRANT CREATE SYNONYM TO SIGINPI;
  GRANT CREATE VIEW TO SIGINPI;
  GRANT UNLIMITED TABLESPACE TO SIGINPI;
  -- 2 Tablespace Quotas for SIGINPI 
  ALTER USER SIGINPI QUOTA UNLIMITED ON SIGINPI_IND;
  ALTER USER SIGINPI QUOTA UNLIMITED ON SIGINPI_TAB;
  -- 1 Object Privilege for SIGINPI 
    GRANT SELECT ON SYS.V_$INSTANCE TO SIGINPI;
