2) USUARIOS

CREATE USER "AMBGOLDEN"  PROFILE "DEFAULT" 
    IDENTIFIED BY "AMBGOLDEN" DEFAULT TABLESPACE "CENTRAL_TABLES"
    TEMPORARY TABLESPACE "TEMP" 
    QUOTA UNLIMITED 
    ON "CENTRAL_INDEX" 
    QUOTA UNLIMITED 
    ON "CENTRAL_TABLES" 
    ACCOUNT UNLOCK;
GRANT UNLIMITED TABLESPACE TO "AMBGOLDEN";
GRANT "CONNECT" TO "AMBGOLDEN";
GRANT "RESOURCE" TO "AMBGOLDEN";
