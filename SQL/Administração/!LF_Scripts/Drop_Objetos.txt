object_type || ' ' ||
owner || '.' ||
object_name ||
DECODE (object_type, 'TABLE', ' cascade constraints;', ';')
FROM all_objects
WHERE object_type not in ('TRIGGER', 'PACKAGE BODY', 'INDEX', 'LOB')
AND owner = 'POUSADA'
