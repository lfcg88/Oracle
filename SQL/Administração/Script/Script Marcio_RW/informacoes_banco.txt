select sum(((bytes/1024)/1024)/1024) from dba_data_files;
//tamanho do banco em Gb

select tablespace_name, sum((bytes/1024)/1024) from dba_data_files
group by tablespace_name;
//tamanho das tablespaces em Mb

select tablespace_name, sum(((bytes/1024)/1024)/1024) from dba_temp_files
group by tablespace_name;
// tamanho das tablespaces temporarias em Mb
