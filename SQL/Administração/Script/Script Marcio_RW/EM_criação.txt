#### Recriar o EM Standalone #####
emca -config dbcontrol db -repos recreate -silent -ORACLE_HOSTNAME unimedor03 -SID unimed3 -PORT 1521 -ORACLE_HOME $ORACLE_HOME -DBSNMP_PWD unimed -SYSMAN_PWD unimed -SYS_PWDunimedinfo

#### Dropar o EM Standalone #####
emca -deconfig dbcontrol db -repos drop

#### Criar EM Standalone ###
emca -config dbcontrol db -repos create

### Configurar EM como Não-Seguro - Standalone/Cluster ###
emctl unsecure dbconsole

#### Reconfigurar EM Cluster ####
emca -reconfig dbcontrol –cluster –EM_NODE unimedor03 -EM_SID_LIST unimed3,unimed4

#### Dropar EM Cluster ####
emca -repos drop -cluster -DB_UNIQUE_NAME unimed -PORT 1521 -SYS_PWD unimedinfo -DBSNMP_PWD unimed -SYSMAN_PWD unimed -SERVICE_NAME unimedrac
